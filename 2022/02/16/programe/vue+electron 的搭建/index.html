# 2. 最简单的 electron 渲染进程 与 主进程的通信

在渲染进程 About.vue

<template>
  <div class="about">
    <input id="name">
    <p id="result" color='black'></p>
    <button @click="openfile">打开文件</button>
  </div>
</template>
<script>
export default {
  data () {
    return {
      name: 'hello'
    }
  },
  methods: {
    openfile () {
      const ipc = require('electron').ipcRenderer
      ipc.send('open-dialog') // 向主进程发送事件
    }
  },
  created () {
    const ipc = require('electron').ipcRenderer  
    ipc.on('open-finished', function (event, filepath) {  //异步监听主进程发来的消息
      console.log(filepath)
    })
  }
}
</script>
